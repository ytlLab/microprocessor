## 元件介紹
MPU6050 是由 InvenSense 公司推出的全球首款整合型 6 軸運動處理組件。在電子愛好者、機器人開發以及無人機領域非常受歡迎，主要是因為其高性價比、體積小巧且功能強大。</br>

它在一個晶片中結合了 3 軸陀螺儀（Gyroscope） 與 3 軸加速度計（Accelerometer），並內建了一個 數字運動處理器（DMP, Digital Motion Processor）。</br>
可量測
- 三軸加速度
- 三軸角速度
- 溫度

| 引腳名稱 | 功能描述 | 
| :---: | :---: | 
|VCC	|電源輸入（通常支援 3.3V 至 5V）|
|GND	|接地|
|SCL	|I2C 時脈線|
|SDA	|I2C 數據線|
|XDA / XCL	|輔助 I2C（用於連接外部感測器，如磁力計）|
|AD0	|I2C 地址選擇位（接地為 0x68，接高電位為 0x69）|
|INT	|中斷輸出引腳（數據準備好時通知主控板）|

## 範例 1 輸出序列埠
```
//ch07_01
#include <Adafruit_MPU6050.h>  // 引入Adafruit_MPU6050函式庫，用於控制MPU6050

Adafruit_MPU6050 mpu;

void setup() {
  Serial.begin(115200);

  if (!mpu.begin()) {                    // 初始化mpu，若失敗則顯示文字
    Serial.println("MPU6050 not found");
    while (1) delay(10);
  }

  // 設定量測範圍
  mpu.setAccelerometerRange(MPU6050_RANGE_8_G); // 量測「加速度」，ex：2、4、8、16
  mpu.setGyroRange(MPU6050_RANGE_500_DEG);      // 量測「轉動的速度」，ex：250、500、1000、2000
  mpu.setFilterBandwidth(MPU6050_BAND_21_HZ);   // 設定低通濾波器頻寬為21Hz，用於去除高頻雜訊
}

void loop() {
  sensors_event_t a, g, temp;       // 用來儲存感測器回傳的資料 
  mpu.getEvent(&a, &g, &temp);      // &=取址符號，把讀到的資料寫入對應的變數位址 

  // 輸出加速度
  Serial.print("Acc (m/s^2) X:");
  Serial.print(a.acceleration.x);   // 讀取加速度X軸的值
  Serial.print(" Y:");
  Serial.print(a.acceleration.y);   // 讀取加速度Y軸的值
  Serial.print(" Z:");
  Serial.println(a.acceleration.z); // 讀取加速度Z軸的值

  // 輸出陀螺儀
  Serial.print("Gyro (rad/s) X:");
  Serial.print(g.gyro.x);           // 讀取角速度X軸的值
  Serial.print(" Y:");              
  Serial.print(g.gyro.y);           // 讀取角速度Y軸的值
  Serial.print(" Z:");
  Serial.println(g.gyro.z);         // 讀取角速度Z軸的值

  // 輸出溫度
  Serial.print("Temp: ");       
  Serial.print(temp.temperature);   // 讀取溫度的值
  Serial.println(" °C");

  Serial.println();
  delay(1000);
}
```
## 範例 2 輸出LCD1602

> 提醒，需安裝 LiquidCrystal_I2C-master 程式庫</br>

```
//ch07_02

#include <Adafruit_MPU6050.h>       // 引入Adafruit_MPU6050函式庫，用於控制MPU6050
#include <LiquidCrystal_I2C.h>      // 引入LiquidCrystal_I2C函式庫，用於控制I2C介面的LCD顯示器
LiquidCrystal_I2C lcd(0x27,16,2);   // 創建LCD物件，設定I2C地址為0x27，並指定 LCD為16列(characters)2行(rows)

Adafruit_MPU6050 mpu;

void setup() {
  Serial.begin(115200);
  while (!Serial) delay(10);             // 等Serial初始化完成才開始輸出資料，避免漏掉輸出資訊

  if (!mpu.begin()) {                    // 如果感測器初始化失敗，則顯示錯誤
    Serial.println("MPU6050 not found");
    while (1) delay(10);
  }

  // 設定量測範圍
  mpu.setAccelerometerRange(MPU6050_RANGE_8_G); // 量測「多大的動作」，ex：2、4、8、16
  mpu.setGyroRange(MPU6050_RANGE_500_DEG);      // 量測「轉動的速度」，ex：250、500、1000、2000
  mpu.setFilterBandwidth(MPU6050_BAND_21_HZ);   // 設定低通濾波器頻寬為21Hz，用於去除高頻雜訊
  
  lcd.init();                   // 初始化LCD顯示器
  lcd.backlight();              // 開啟LCD背光     
    
  delay(100);
}

void loop() {
  sensors_event_t a, g, temp;
  mpu.getEvent(&a, &g, &temp);

  // 輸出加速度
  Serial.print("Acc (m/s^2) X:");
  Serial.print(a.acceleration.x);
  Serial.print(" Y:");
  Serial.print(a.acceleration.y);
  Serial.print(" Z:");
  Serial.println(a.acceleration.z);
  lcd.setCursor(0,0);           
  lcd.print("X:");              
  lcd.setCursor(2,0);           
  lcd.print(a.acceleration.x);  
  lcd.setCursor(6,0);           
  lcd.print("Y:");             
  lcd.setCursor(8,0);           
  lcd.print(a.acceleration.y);  
  lcd.setCursor(12,0);          
  lcd.print("Z:");            
  lcd.setCursor(14,0);          
  lcd.print(a.acceleration.z);  

  // 輸出陀螺儀
  Serial.print("Gyro (rad/s) X:");
  Serial.print(g.gyro.x);
  Serial.print(" Y:");
  Serial.print(g.gyro.y);
  Serial.print(" Z:");
  Serial.println(g.gyro.z);
  lcd.setCursor(0,1);           
  lcd.print("x:");              
  lcd.setCursor(2,1);           
  lcd.print(g.gyro.x);  
  lcd.setCursor(6,1);           
  lcd.print("y:");             
  lcd.setCursor(8,1);           
  lcd.print(g.gyro.y);  
  lcd.setCursor(12,1);          
  lcd.print("z:");            
  lcd.setCursor(14,1);          
  lcd.print(g.gyro.z); 


  // 輸出溫度
  Serial.print("Temp: ");
  Serial.print(temp.temperature);
  Serial.println(" °C");

  Serial.println();
  delay(100);
}
```

```
      }
   }
  }
}
```
