接線方式

1. LED長腳 -> GPIO 13(S)
2. LED短腳 -> 電阻(220Ω) -> GND
3. 按鈕1(左)-> 3.3v
4. 按鈕1(右)-> GPIO 25(S) ->電阻(10k)-> GND

#### if-else 條件判斷

<img width="800" height="400" alt="image" src="https://github.com/user-attachments/assets/122fe985-62fa-4d06-a1f0-3859731398ff" />

```
//ch02_ex03

# define LED_IN 2          // 將內鍵LED定義為GPIO 2
# define LED 13            // 將LED定義為GPIO 13
# define button1 25        // 將button1定義為GPIO 25
# define button2 26        // 將button2定義為GPIO 26

void setup()
{
  pinMode(LED_IN,OUTPUT);  // LED_IN腳位，設定為OUTPUT
  pinMode(LED,OUTPUT);     // LED腳位，設定為OUTPUT
  pinMode(button1,INPUT);  // button1腳位，設定為INPUT
  pinMode(button2,INPUT);  // button2腳位，設定為INPUT
}
void loop()
{
 
  if( digitalRead(button1) == HIGH ){        
      digitalWrite(LED_IN,HIGH);     
  }
  else{                            
      digitalWrite(LED_IN,LOW); 
  }





}
```
