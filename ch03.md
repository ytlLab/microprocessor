#define LED_R 12         // 將LED_R 定義為GPIO 12

#define LED_G 27         // 將LED_G 定義為GPIO 27

  // 初始化所有 LED 接腳為輸出模式
  pinMode(LED_R, OUTPUT);
  pinMode(LED_Y, OUTPUT);
  pinMode(LED_G, OUTPUT);
}

void loop()
{
  // 1. 紅燈亮 1 秒
  digitalWrite(LED_R, HIGH);
  digitalWrite(LED_Y, LOW);
  digitalWrite(LED_G, LOW);
  delay(1000);

  // 2. 黃燈亮 1 秒
  digitalWrite(LED_R, LOW);
  digitalWrite(LED_Y, HIGH);
  digitalWrite(LED_G, LOW);
  delay(1000);

  // 3. 綠燈亮 1 秒
  digitalWrite(LED_R, LOW);
  digitalWrite(LED_Y, LOW);
  digitalWrite(LED_G, HIGH);
  delay(1000);
}

## 結合按鈕功能
程式運作流程</br>
1. 偵測按鈕是否被按下 (高電位輸入)
  - 先判斷 按鈕1是否為高電位，代表按鈕被按下。

2. 根據目前狀態 (state) 進行切換 </br>
  - 若目前 state = 1，按下按鈕後 state +1，變成 state = 2。
  - 若目前 state = 2，按下按鈕後 state +1，變成 state = 3。
  - 若目前不是 1 或 2 (例如: state = 3 或其他數值)，則重置回 state = 1。

3. 按鈕2的邏輯與按鈕1相同</br>
- 可獨立控制另一組 LED 燈。</br>

4. 根據 state 數值改變 LED 燈光狀態</br>
- state = 1 → 紅燈亮
- state = 2 → 黃燈亮
- state = 3 → 綠燈亮

## 程式範例概念
```
if(按鈕1是否為高電位){
    if (state == 1){
        state = state + 1;
    }
    else if (state == 2){
        state = state +1;
    }
    else{
       state = 1;
    }
}

## 練習1
每次亮一顆燈，</br>
點擊右邊按鈕，燈號右移；</br>
點擊左邊按鈕，燈號左移。</br>
```
//ch03_02
#define LED_R 12
#define LED_Y 14
#define LED_G 27
#define BTN_L 26  // 定義左邊按鈕接腳 (使用 GPIO 26)
#define BTN_R 25  // 定義右邊按鈕接腳 (使用 GPIO 25)

int state = 1;    // 1:紅, 2:黃, 3:綠

void setup() {
  pinMode(LED_R, OUTPUT);
  pinMode(LED_Y, OUTPUT);
  pinMode(LED_G, OUTPUT);
  
  // 設定按鈕為輸入模式
  pinMode(BTN_L, INPUT); 
  pinMode(BTN_R, INPUT);
}

void loop() {
  // --- 按鈕邏輯處理 ---
  
  // 點擊右邊按鈕：燈號右移 (R -> Y -> G)
  if (digitalRead(BTN_R) == HIGH) {
    state++;
    if (state > 3) state = 1; // 超過綠燈就回到紅燈
    delay(200); // 簡單的防彈跳延時
  }

  // 點擊左邊按鈕：燈號左移 (G -> Y -> R)
  if (digitalRead(BTN_L) == HIGH) {
    state--;
    if (state < 1) state = 3; // 低於紅燈就回到綠燈
    delay(200);
  }

  // --- 燈號顯示邏輯 ---
  



  
}

## 程式範例
```
//ch02_03
# define trig 33        // 將trig定義為GPIO 33 
# define echo 32        // 將echo定義為GPIO 32 

void setup() {
  pinMode(trig, OUTPUT);   // trig腳位，設定為OUTPUT (發出聲波)
  pinMode(echo, INPUT);    // echo腳位，設定為INPUT (接收聲波)
  Serial.begin(115200);    // 初始化序列通訊界面，鮑率設為115200
}

void loop() {
  double d = ping()/58;    // 計算有多少個 單程1cm的時間
  Serial.print(d);         // 顯示傳回的時間
  Serial.println("cm");    
  delay(100);
}

double ping() { 
  digitalWrite(trig, HIGH);       // 發出超音波
  delayMicroseconds(10);          // 持續10微秒 sustain at least 10us HIGH pulse
  digitalWrite(trig, LOW);        // 關閉超音波
  return pulseIn(echo, HIGH);     // 計算超音波傳回的時間
 }

```
## 序列埠的使用與設定
<img width="800" height="400" alt="image" src="https://github.com/user-attachments/assets/f95e8569-ae9f-459e-8247-f98c0870186f" />

### Serial.begin( )的作用

- Serial.begin(115200) 是用來初始化 Arduino 與電腦間的序列通訊介面，使用序列埠時需撰寫。
- 參數(115200)是鮑率(Baud Rate)，表示每秒傳輸的位元數(bits per second)。
- 常搭配 Serial.print()、Serial.println() 使用

<img width="400" height="150" alt="image" src="https://github.com/user-attachments/assets/aff6ee25-382f-4b24-928f-1c523ded0aab" />

### pulseIn(echo, HIGH)的作用
- pulseIn(echo, HIGH) 會等到 Echo腳位變成高電位(HIGH)開始計時，一旦 Echo腳位變回低電位(LOW)則計時結束，最後回傳的數值就是聲波往返所花的時間(單位：微秒 us)。
- 發射聲波後，會自動變成高電位(HIGH)
- 當Echo腳位偵測到反射回來的聲波時，Echo腳位會立刻變回低電位(LOW)
