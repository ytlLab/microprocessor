## 範例程式
```
//ch04_01
#include <LiquidCrystal_I2C.h>      // 引入LiquidCrystal_I2C函式庫，用於控制I2C介面的LCD顯示器
LiquidCrystal_I2C lcd(0x27,16,2);   // 創建LCD物件，設定I2C地址為0x27，並指定 LCD為16列(characters)2行(rows)
int count = 0; 

void setup()
{
  lcd.init();                   // 初始化LCD顯示器
  lcd.backlight();              // 開啟LCD背光       
}

void loop()
{
  lcd.setCursor(2,0);           // 設定游標位置至(2,0)第一行第3格開始
  lcd.print("Hello, Wego!");    // 顯示 "Hello, Wego!"
  lcd.setCursor(0,1);           // 設定游標位置至(0,1)第二行第1格開始
  lcd.print("Count:");          // 顯示 "Count:"
  lcd.setCursor(6,1);           // 設定游標位置至(6,1)第二行第7格開始
  lcd.print(count);             // 顯示count值
  delay(1000);
  count++;
}

```
