## 範例
```
//ch05_01
#include <Keypad.h>         // 鍵盤函式庫

const byte ROWS = 4;        // 宣告列為4，const為不能被更改，byte數值範圍0~255，占用1 byte記憶體
const byte COLS = 4;        // 宣告行為4，const為不能被更改，byte數值範圍0~255，占用1 byte記憶體
char keys[ROWS][COLS] = {   // 鍵盤對應字元
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};
byte rowPins[ROWS] = {0,4,16,17};   // row接腳
byte colPins[COLS] = {5,18,19,21};  // col接腳
Keypad keynum = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS );  // 創建Keypad物件叫做keynum，並放入上述的資料

void setup(){
  Serial.begin(115200);
}

void loop(){
  char key = keynum.getKey();        // 讀取按鍵值並存入key
  if (key != NO_KEY){                // 判斷是否有按鍵值
    Serial.println(key);             // 在序列埠監控視窗顯示按鍵值
  }
}
```
